```prisma
// This is your Prisma schema file,
// learn more about it in the docs: https://pris.ly/d/prisma-schema

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "mysql"

// ================================
// USER MANAGEMENT
// ================================

model User {
  id          String   @id @default(cuid())
  email       String   @unique
  firstName   String
  lastName    String
  password    String?  // Nullable for OAuth users
  role        UserRole @default(EMPLOYEE)
  isActive    Boolean  @default(true)
  lastLoginAt DateTime?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // Relations
  company   Company @relation(fields: [companyId], references: [id], onDelete: Cascade)
  companyId String

  createdReports  Report[]       @relation("ReportCreator")
  assignedTasks   Task[]
  auditLogs       AuditLog[]
  notifications   Notification[]
  surveyResponses SurveyResponse[]

  @@map("users")
}

enum UserRole {
  SUPER_ADMIN
  ADMIN
  MANAGER
  EMPLOYEE
  VIEWER
}

// ================================
// COMPANY MANAGEMENT
// ================================

model Company {
  id          String      @id @default(cuid())
  name        String
  nip         String      @unique
  regon       String?
  industry    String
  size        CompanySize
  website     String?
  description String?     @db.Text
  isActive    Boolean     @default(true)
  createdAt   DateTime    @default(now())
  updatedAt   DateTime    @updatedAt

  // Address
  street     String
  city       String
  postalCode String
  country    String @default("Poland")

  // Relations
  users       User[]
  esgMetrics  EsgMetric[]
  reports     Report[]
  policies    Policy[]
  surveys     Survey[]
  tasks       Task[]
  auditLogs   AuditLog[]
  riskItems   RiskItem[]

  @@map("companies")
}

enum CompanySize {
  MICRO      // < 10 employees
  SMALL      // 10-49 employees
  MEDIUM     // 50-249 employees
  LARGE      // 250+ employees
}

// ================================
// ESG METRICS
// ================================

model EsgMetric {
  id          String      @id @default(cuid())
  category    EsgCategory
  type        String      // e.g., "carbon_emissions", "energy_consumption"
  name        String
  value       Float
  unit        String
  period      String      // e.g., "2024-Q1", "2024-01"
  source      String?     // Data source
  verified    Boolean     @default(false)
  notes       String?     @db.Text
  createdAt   DateTime    @default(now())
  updatedAt   DateTime    @updatedAt

  // Relations
  company   Company @relation(fields: [companyId], references: [id], onDelete: Cascade)
  companyId String

  @@unique([companyId, category, type, period])
  @@index([companyId, category])
  @@index([createdAt])
  @@map("esg_metrics")
}

enum EsgCategory {
  ENVIRONMENTAL
  SOCIAL
  GOVERNANCE
}

// ================================
// REPORTING
// ================================

model Report {
  id          String     @id @default(cuid())
  title       String
  type        ReportType
  framework   String?    // GRI, SASB, TCFD, etc.
  period      String     // e.g., "2024-Q1"
  status      ReportStatus @default(DRAFT)
  content     Json?      // Report content as JSON
  fileUrl     String?    // Generated file URL
  publishedAt DateTime?
  createdAt   DateTime   @default(now())
  updatedAt   DateTime   @updatedAt

  // Relations
  company   Company @relation(fields: [companyId], references: [id], onDelete: Cascade)
  companyId String
  
  createdBy User   @relation("ReportCreator", fields: [createdById], references: [id])
  createdById String

  @@index([companyId, type])
  @@index([status])
  @@map("reports")
}

enum ReportType {
  SUSTAINABILITY
  ESG_SCORECARD
  CARBON_FOOTPRINT
  DIVERSITY_INCLUSION
  GOVERNANCE
  CUSTOM
}

enum ReportStatus {
  DRAFT
  REVIEW
  APPROVED
  PUBLISHED
  ARCHIVED
}

// ================================
// SURVEYS
// ================================

model Survey {
  id          String        @id @default(cuid())
  title       String
  description String?       @db.Text
  type        SurveyType
  status      SurveyStatus  @default(DRAFT)
  startDate   DateTime?
  endDate     DateTime?
  isAnonymous Boolean       @default(true)
  createdAt   DateTime      @default(now())
  updatedAt   DateTime      @updatedAt

  // Relations
  company   Company @relation(fields: [companyId], references: [id], onDelete: Cascade)
  companyId String

  questions SurveyQuestion[]
  responses SurveyResponse[]

  @@index([companyId, status])
  @@map("surveys")
}

enum SurveyType {
  EMPLOYEE_SATISFACTION
  DIVERSITY_INCLUSION
  WORKPLACE_SAFETY
  ETHICS_COMPLIANCE
  CUSTOM
}

enum SurveyStatus {
  DRAFT
  ACTIVE
  COMPLETED
  ARCHIVED
}

model SurveyQuestion {
  id       String       @id @default(cuid())
  question String       @db.Text
  type     QuestionType
  options  Json?        // For multiple choice questions
  required Boolean      @default(false)
  order    Int

  // Relations
  survey   Survey @relation(fields: [surveyId], references: [id], onDelete: Cascade)
  surveyId String

  responses SurveyResponse[]

  @@index([surveyId, order])
  @@map("survey_questions")
}

enum QuestionType {
  TEXT
  TEXTAREA
  RADIO
  CHECKBOX
  SELECT
  SCALE
  DATE
}

model SurveyResponse {
  id       String @id @default(cuid())
  answer   Json   // Response data
  createdAt DateTime @default(now())

  // Relations
  survey     Survey         @relation(fields: [surveyId], references: [id], onDelete: Cascade)
  surveyId   String
  
  question   SurveyQuestion @relation(fields: [questionId], references: [id], onDelete: Cascade)
  questionId String
  
  user       User?          @relation(fields: [userId], references: [id], onDelete: SetNull)
  userId     String?        // Nullable for anonymous responses

  @@unique([surveyId, questionId, userId])
  @@index([surveyId])
  @@map("survey_responses")
}

// ================================
// POLICIES & COMPLIANCE

model Policy {

  title       String
  category    PolicyCategory
  content     String       @db.LongText
  version     String       @default("1.0")
  status      PolicyStatus @default(DRAFT)
  effectiveDate DateTime?
  reviewDate  DateTime?
  createdAt   DateTime     @default(now())
  updatedAt   DateTime     @updatedAt

  // Relations
  company   Company @relation(fields: [companyId], references: [id], onDelete: Cascade)
  companyId String

  @@index([companyId, category])
  @@index([status])
  @@map("policies")
}

enum PolicyCategory {
  CODE_OF_CONDUCT
  ENVIRONMENTAL
  HEALTH_SAFETY
  DIVERSITY_INCLUSION
  DATA_PRIVACY
  ANTI_CORRUPTION
  SUPPLY_CHAIN
  HUMAN_RIGHTS

enum PolicyStatus {
  DRAFT
  REVIEW
  APPROVED
  ACTIVE
  ARCHIVED
}

// ================================
// RISK MANAGEMENT
// ================================

model RiskItem {
  id          String     @id @default(cuid())
  title       String
  description String     @db.Text
  category    RiskCategory
  probability RiskLevel
  impact      RiskLevel
  riskScore   Float      // Calculated: probability * impact
  status      RiskStatus @default(IDENTIFIED)
  owner       String?    // Risk owner
  mitigation  String?    @db.Text
  dueDate     DateTime?
  createdAt   DateTime   @default(now())
  updatedAt   DateTime   @updatedAt

  // Relations
  company   Company @relation(fields: [companyId], references: [id], onDelete: Cascade)
  companyId String

  @@index([companyId, category])
  @@index([riskScore])
  @@map("risk_items")
}

enum RiskCategory {
  ENVIRONMENTAL
  SOCIAL
  GOVERNANCE
  OPERATIONAL
  FINANCIAL
  REGULATORY
  REPUTATIONAL
}

enum RiskLevel {
  LOW
  MEDIUM
  HIGH
  CRITICAL
}

enum RiskStatus {
  IDENTIFIED
  ASSESSED
  MITIGATED
  MONITORED
  CLOSED
}

// ================================
// TASK MANAGEMENT
// ================================

model Task {
  id          String     @id @default(cuid())
  title       String
  description String?    @db.Text
  priority    TaskPriority @default(MEDIUM)
  status      TaskStatus @default(TODO)
  dueDate     DateTime?
  completedAt DateTime?
  createdAt   DateTime   @default(now())
  updatedAt   DateTime   @updatedAt

  // Relations
  company   Company @relation(fields: [companyId], references: [id], onDelete: Cascade)
  companyId String
  
  assignedTo User    @relation(fields: [assignedToId], references: [id])
  assignedToId String

  @@index([companyId, status])
  @@index([assignedToId])
  @@index([dueDate])
  @@map("tasks")
}

enum TaskPriority {
  LOW
  MEDIUM
  HIGH
  URGENT
}

enum TaskStatus {
  TODO
  IN_PROGRESS
  REVIEW
  COMPLETED
  CANCELLED
}

// ================================
// NOTIFICATIONS
// ================================

model Notification {
  id        String           @id @default(cuid())
  title     String
  message   String           @db.Text
  type      NotificationType
  priority  NotificationPriority @default(MEDIUM)
  read      Boolean          @default(false)
  readAt    DateTime?
  createdAt DateTime         @default(now())

  // Relations
  user   User   @relation(fields: [userId], references: [id], onDelete: Cascade)
  userId String

  @@index([userId, read])
  @@index([createdAt])
  @@map("notifications")
}

enum NotificationType {
  TASK_ASSIGNED
  TASK_DUE
  REPORT_READY
  SURVEY_AVAILABLE
  POLICY_UPDATED
  RISK_ALERT
  SYSTEM_UPDATE
}

enum NotificationPriority {
  LOW
  MEDIUM
  HIGH
  URGENT
}

// ================================
// AUDIT LOG
// ================================

model AuditLog {
  id        String    @id @default(cuid())
  action    String    // e.g., "CREATE", "UPDATE", "DELETE"
  entity    String    // e.g., "User", "Report", "Policy"
  entityId  String    // ID of the affected entity
  oldValues Json?     // Previous values
  newValues Json?     // New values
  ipAddress String?
  userAgent String?
  createdAt DateTime  @default(now())

  // Relations
  company   Company @relation(fields: [companyId], references: [id], onDelete: Cascade)
  companyId String
  
  user      User    @relation(fields: [userId], references: [id])
  userId    String

  @@index([companyId, entity])
  @@index([userId])
  @@index([createdAt])
  @@map("audit_logs")
}

// ================================
// SYSTEM CONFIGURATION
// ================================

model SystemConfig {
  id    String @id @default(cuid())
  key   String @unique
  value Json
  updatedAt DateTime @updatedAt

  @@map("system_config")
}
```

### Krok 3: Backup Strategy

#### backup-script.sh
```bash
#!/bin/bash

# Database backup script
set -e

# Configuration
DB_NAME="esg_platform_prod"
DB_USER="esg_prod"
DB_PASSWORD="secure_prod_password"
DB_HOST="mysql-prod.cyberfolks.pl"
BACKUP_DIR="/backups/mysql"
DATE=$(date +"%Y%m%d_%H%M%S")
BACKUP_FILE="$BACKUP_DIR/${DB_NAME}_${DATE}.sql"

# Create backup directory
mkdir -p $BACKUP_DIR

# Create backup
echo "üîÑ Creating database backup..."
mysqldump \
  --host=$DB_HOST \
  --user=$DB_USER \
  --password=$DB_PASSWORD \
  --single-transaction \
  --routines \
  --triggers \
  --events \
  --hex-blob \
  --default-character-set=utf8mb4 \
  $DB_NAME > $BACKUP_FILE

# Compress backup
echo "üì¶ Compressing backup..."
gzip $BACKUP_FILE

# Upload to cloud storage (optional)
if [ ! -z "$AWS_S3_BUCKET" ]; then
  echo "‚òÅÔ∏è Uploading to S3..."
  aws s3 cp "${BACKUP_FILE}.gz" "s3://$AWS_S3_BUCKET/database-backups/"
fi

# Clean old backups (keep last 30 days)
echo "üßπ Cleaning old backups..."
find $BACKUP_DIR -name "*.sql.gz" -mtime +30 -delete

echo "‚úÖ Backup completed: ${BACKUP_FILE}.gz"
```

```sql
-- Performance monitoring queries

-- 1. Connection status
SELECT 
  VARIABLE_NAME,
  VARIABLE_VALUE
FROM performance_schema.global_status
WHERE VARIABLE_NAME IN (
  'Threads_connected',
  'Threads_running',
  'Max_used_connections',
  'Aborted_connects'
);

-- 2. Query performance
SELECT 
  digest_text,
  count_star,
  avg_timer_wait/1000000000 as avg_time_ms,
  max_timer_wait/1000000000 as max_time_ms,
  sum_rows_examined/count_star as avg_rows_examined
FROM performance_schema.events_statements_summary_by_digest
ORDER BY avg_timer_wait DESC
LIMIT 10;

-- 3. Table access patterns
SELECT 
  object_schema,
  object_name,
  count_read,
  count_write,
  count_fetch,
  count_insert,
  count_update,
  count_delete
FROM performance_schema.table_io_waits_summary_by_table
WHERE object_schema = 'esg_platform_prod'
ORDER BY count_read + count_write DESC;

-- 4. Index usage
SELECT 
  object_schema,
  object_name,
  index_name,
  count_fetch,
  count_insert,
  count_update,
  count_delete
FROM performance_schema.table_io_waits_summary_by_index_usage
WHERE object_schema = 'esg_platform_prod'
ORDER BY count_fetch DESC;

-- 5. Lock waits
SELECT 
  object_schema,
  object_name,
```

```bash
#!/bin/bash

# Database maintenance script
set -e

DB_NAME="esg_platform_prod"
DB_USER="esg_prod"
DB_PASSWORD="secure_prod_password"
DB_HOST="mysql-prod.cyberfolks.pl"

echo "üîß Starting database maintenance..."

# 1. Optimize tables
echo "‚ö° Optimizing tables..."
mysql -h $DB_HOST -u $DB_USER -p$DB_PASSWORD -e "
USE $DB_NAME;
OPTIMIZE TABLE users;
OPTIMIZE TABLE companies;
OPTIMIZE TABLE esg_metrics;
OPTIMIZE TABLE reports;
OPTIMIZE TABLE audit_logs;
"

# 2. Update table statistics
echo "üìä Updating table statistics..."
mysql -h $DB_HOST -u $DB_USER -p$DB_PASSWORD -e "
USE $DB_NAME;
ANALYZE TABLE users;
ANALYZE TABLE companies;
ANALYZE TABLE esg_metrics;
ANALYZE TABLE reports;
ANALYZE TABLE audit_logs;
"

# 3. Clean old audit logs (keep last 90 days)
echo "üßπ Cleaning old audit logs..."
mysql -h $DB_HOST -u $DB_USER -p$DB_PASSWORD -e "
USE $DB_NAME;
DELETE FROM audit_logs 
WHERE created_at < DATE_SUB(NOW(), INTERVAL 90 DAY);
"

# 4. Clean old notifications (keep last 30 days)
echo "üßπ Cleaning old notifications..."
mysql -h $DB_HOST -u $DB_USER -p$DB_PASSWORD -e "
USE $DB_NAME;
DELETE FROM notifications 
WHERE created_at < DATE_SUB(NOW(), INTERVAL 30 DAY) AND read = 1;
"

# 5. Check for fragmentation
echo "üîç Checking table fragmentation..."
mysql -h $DB_HOST -u $DB_USER -p$DB_PASSWORD -e "
SELECT 
  table_name,
  ROUND(data_length/1024/1024, 2) as data_mb,
  ROUND(data_free/1024/1024, 2) as free_mb,
  ROUND(data_free/data_length*100, 2) as fragmentation_percent
FROM information_schema.tables
WHERE table_schema = '$DB_NAME'
AND data_free > 0
ORDER BY fragmentation_percent DESC;
"

echo "‚úÖ Database maintenance completed!"
```

```bash
#!/bin/bash

# Database deployment script
set -e

ENVIRONMENT=${1:-development}

echo "üöÄ Deploying database to $ENVIRONMENT environment..."

case $ENVIRONMENT in
  "development")
    echo "üîß Setting up development database..."
    docker-compose up -d mysql
    sleep 30
    
    # Run migrations
    npx prisma migrate dev
    
    # Seed database
    npx prisma db seed
    ;;
    
  "staging")
    echo "üîß Setting up staging database..."
    # Run migrations
    npx prisma migrate deploy
    
    # Optional: seed with test data
    if [ "$SEED_STAGING" = "true" ]; then
      npx prisma db seed
    fi
    ;;
    
  "production")
    echo "üîß Setting up production database..."
    # Backup current database
    ./scripts/backup-database.sh
    
    # Run migrations
    npx prisma migrate deploy
    
    # Verify deployment
    npx prisma validate
    ;;
    
  *)
    echo "‚ùå Unknown environment: $ENVIRONMENT"
    echo "Available environments: development, staging, production"
    exit 1
    ;;
esac

echo "‚úÖ Database deployment to $ENVIRONMENT completed!"
```
